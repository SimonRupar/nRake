  def attr_array(*names)
    names.each do |n|
      self.send :define_method, n do |*value|
        if value.nil? || value.empty?
          return instance_variable_get "@#{n}"
        else
          instance_variable_set "@#{n}", value
        end
      end
      self.send :define_method, "#{n}=" do |value|
        instance_variable_set "@#{n}", value
      end
    end
  end
